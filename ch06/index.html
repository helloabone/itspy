    # 從 datetime 模組匯入 datetime 類別
# 並為 datetime 類別設定別名 dt
# 後續用別名操作方法就行
from datetime import datetime as dt

# 將今天日期時間存入 today 變數
today = dt.today()
# 輸出今天日期時間
print(today)
# 輸出年
print(today.year)
# 輸出月
print(today.month)
# 輸出日
print(today.day)

    import math

def float_to_abs(x):
    print('(1) x:', x)
    # 取浮點數的絕對值
    # x = abs(x)
    x = math.fabs(x)
    print('(2) x:', x)
    # 無條件進位到整數
    x = math.ceil(x)
    print('(3) x:', x)

float_to_abs(-12.48)
